{\bfseries{STASH VCS}} is an educational project implementing a version control system. ~\newline
 The project is developed in C++ using CMake for building and Doxygen for documentation.

\DoxyHorRuler{0}
\hypertarget{md__r_e_a_d_m_e_autotoc_md36}{}\doxysection{\texorpdfstring{Key Features ⚙️}{Key Features ⚙️}}\label{md__r_e_a_d_m_e_autotoc_md36}

\begin{DoxyItemize}
\item {\bfseries{Branch Management\+:}} Create, switch, and delete branches.
\item {\bfseries{\doxylink{struct_commit}{Commit} Changes\+:}} Track and save changes in the repository. {\itshape (soon)}
\item {\bfseries{Branch Merging\+:}} Combine changes from different branches. {\itshape (soon)}
\item {\bfseries{Revert Changes\+:}} Roll back to previous versions of files. {\itshape (soon)}
\item {\bfseries{Logging\+:}} View the history of changes. {\itshape (soon)}
\end{DoxyItemize}
